for i in df.columns:
    mn = df[i].isna().sum()
    if mn < 30000 and df[i].dtype == 'object':
        df[i] = df[i].fillna(df[i].mode()[0])
    elif mn < 30000 and df[i].dtype != 'object':
        df[i] = df[i].fillna(df[i].mean())
    else:
        df.drop(columns=i, inplace=True)
